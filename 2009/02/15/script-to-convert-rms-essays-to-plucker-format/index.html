Script to Convert rms-essays to Plucker Format

15 Feb 2009

!/bin/bash
</br>
</br># Copyright 2009 Rob Myers rob@robmyers.org
</br># Licenced under the GPL 3 or, at your option, any later version.
</br>
</br># Produce a Plucker version of Free Software, Free Society
</br># Some texinfo errors not fixed
</br>
</br># Convert eps images to GIFs
</br>
</br>convert images/clib.eps images/clib.gif
</br>convert images/code.eps images/code.gif
</br>convert images/flex.eps images/flex.gif
</br>convert images/free_software_song.eps images/free_software_song.gif
</br>convert images/headMain.eps images/headMain.gif
</br>convert images/party.eps images/party.gif
</br>convert images/richard.eps images/richard.gif
</br>convert images/philosophical-gnu.eps images/philosophical-gnu.gif
</br>
</br># Fix texinfo problems
</br>
</br>perl -pe ‘s/@heading{(.*)}/@heading $1/’ -i fs_for_freedom.texi
</br>
</br>perl -pe ‘s/^\input texinfo_times.tex//’ <br></br> -i rms-essays.texi
</br>
</br>echo “<br></br>@ifnottex
</br>@alias unnumberedfootnote = footnote
</br>@end ifnottex
</br>
</br>@ifnottex
</br>@macro sp1
</br>@sp 1
</br>@end macro
</br>@end ifnottex
</br>
</br>@include rms-essays.texi
</br>” > rms-essays-html.texi
</br>
</br># Convert to plucker
</br>
</br>makeinfo –html –no-headers –no-split –force -o rms-essays.html <br></br> rms-essays-html.texi
</br>
</br>perl -pe ‘s/^(<a href=”#.+)$/$1<br \/>/’ <br></br> -i rms-essays.html
</br>
</br>plucker-build –zlib-compression –stayonhost –bpp=1 –maxwidth=320 -p . <br></br> -f rms-essays -N “Free Software,Free Society” file://pwd/rms-essays.html
